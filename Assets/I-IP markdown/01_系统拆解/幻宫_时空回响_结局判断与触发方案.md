**三、结局判定流程**

```text
1. 检查是否触发"力量反噬"（优先级最高）
   → 负担 > 70 且未净化？→ 结局 C
   → 禁忌炼金 ≥ 5 次？→ 结局 C
   → 核心碎片 < 3 但强制复活？→ 结局 C

2. 检查最终战选择
   → 选择"接受现实"？→ 结局 D（需记忆碎片 ≥ 5 + 仪式完成）

3. 检查完美重构条件
   → 核心=3 且 记忆≥5 且 负担<30 且 QTE成功 且 疯狂难度？→ 结局 A

4. 其余全部 → 结局 B（按记忆碎片数细分 B1/B2/B3）
   → 记忆碎片 3-4 → B1
   → 记忆碎片 5-6 → B2
   → 高负担净化 → B3
```

// 文件：新建 Assets/Scripts/Systems/EndingSystem.cs
// 实现路径：全新创建，挂载在GameManager上

public enum EndingType { PerfectRebuild, IncompleteReturn, PowerBacklash, SacrificeRedemption, TimeTraveler }

public class EndingSystem : MonoBehaviour:
    // ★ 结局判定需要的全部参数
    [Header("碎片状态")]
    public int coreFragments = 0       // 灵魂核心碎片（0-3）
    public int memoryFragments = 0     // 记忆碎片（0-7）
    
    [Header("负担状态")]
    public float finalBurden = 0       // 最终负担值
    public bool purificationDone = false  // 是否完成净化仪式
    
    [Header("难度")]
    public bool isHardcoreMode = false
    
    [Header("最终战状态")]
    public bool qteSuccess = false     // QTE是否成功
    public bool choseAcceptReality = false  // 是否选择"接受现实"
    
    [Header("统计数据")]
    public int forbiddenAlchemyCount = 0   // 禁忌炼金使用次数
    public int totalCycles = 1             // 总游玩周目
    public int cyclesWithFullFragments = 0 // 累计全碎片周目数
    public bool allRecipesUnlocked = false
    public bool superBeaconCrafted = false
    public bool burdenNeverExceeded20 = false
    
    // ★ 主判定函数
    FUNCTION DetermineEnding() -> EndingType:
        // --- 隐藏结局：时空旅者 ---
        IF totalCycles >= 3 
           AND cyclesWithFullFragments >= 3
           AND allRecipesUnlocked
           AND isHardcoreMode
           AND burdenNeverExceeded20
           AND superBeaconCrafted:
            RETURN TimeTraveler
        
        // --- 优先级1：力量反噬 ---
        IF finalBurden > 70 AND NOT purificationDone:
            RETURN PowerBacklash
        IF forbiddenAlchemyCount >= 5:
            RETURN PowerBacklash
        IF coreFragments < 3 AND NOT choseAcceptReality:
            RETURN PowerBacklash
        
        // --- 优先级2：牺牲救赎 ---
        IF choseAcceptReality AND memoryFragments >= 5 AND purificationDone:
            RETURN SacrificeRedemption
        
        // --- 优先级3：完美重构（真结局）---
        IF coreFragments == 3 
           AND memoryFragments >= 5 
           AND finalBurden < 30 
           AND qteSuccess 
           AND isHardcoreMode:
            RETURN PerfectRebuild
        
        // --- 默认：残缺回归 ---
        RETURN IncompleteReturn
    
    // ★ 结局触发执行
    FUNCTION TriggerEnding(type: EndingType):
        SWITCH type:
            CASE PerfectRebuild:
                PlayCutscene("Ending_PerfectRebuild")
                UnlockAchievement("ACH_TRUE_ENDING")
                UnlockSkin("Sana_TrueForm")
                
            CASE IncompleteReturn:
                // 细分B1/B2/B3
                IF memoryFragments <= 4:
                    PlayCutscene("Ending_B1_Distant")
                ELSE IF memoryFragments <= 6:
                    PlayCutscene("Ending_B2_Confused")
                ELSE:
                    PlayCutscene("Ending_B3_Weakened")
                    
            CASE PowerBacklash:
                PlayCutscene("Ending_PowerBacklash")
                // 主菜单扭曲效果
                PlayerPrefs.SetInt("MainMenuCorrupted", 1)
                
            CASE SacrificeRedemption:
                PlayCutscene("Ending_SacrificeRedemption")
                
            CASE TimeTraveler:
                PlayCutscene("Ending_TimeTraveler")
        
        // 显示结局统计画面
        ShowEndingStatistics()
    
    // ★ 结局UI提示（营火处）
    FUNCTION GetEndingHint() -> string:
        IF coreFragments == 3:
            hint = "莎娜的灵魂核心已完整 ✓"
        ELSE:
            hint = "灵魂核心仍有缺失"
        
        IF memoryFragments >= 5:
            hint += "\n墨语对莎娜的思念足够深刻 ✓"
        
        IF finalBurden > 70:
            hint += "\n⚠ 墨语的灵魂承受着巨大压力"
        ELSE IF finalBurden < 30:
            hint += "\n灵魂纯净，状态极佳 ✓"
        
        IF NOT isHardcoreMode:
            hint += "\n或许更高的难度能解锁更多可能"
        
        RETURN hint

```pseudocode
// 文件：新建 Assets/Scripts/Systems/FinalBattleQTE.cs
// 实现路径：全新创建，在S-SN最终战Phase4触发

public class FinalBattleQTE : MonoBehaviour:
    public int totalRounds = 3          // 3轮QTE
    public int keysPerRound = 5         // 每轮5个按键
    public int maxErrorsAllowed = 2     // 最多2次失误
    public float roundTimeLimit = 5.0   // 每轮限时5秒
    
    int currentRound = 0
    int totalErrors = 0
    bool qteActive = false
    EndingSystem endingSystem
    
    FUNCTION StartQTE():
        qteActive = true
        currentRound = 0
        totalErrors = 0
        PauseEnemyAI()  // Boss暂停攻击
        StartNextRound()
    
    FUNCTION StartNextRound():
        IF currentRound >= totalRounds:
            FinishQTE(true)  // 全部成功
            RETURN
        
        // 生成随机按键序列
        sequence = [RandomKey(), RandomKey(), RandomKey(), RandomKey(), RandomKey()]
        DisplayQTEUI(sequence, roundTimeLimit)
        
        // 等待玩家输入
        StartCoroutine(WaitForQTEInput(sequence))
    
    IEnumerator WaitForQTEInput(sequence: KeyCode[]):
        timeLeft = roundTimeLimit
        inputIndex = 0
        
        WHILE timeLeft > 0 AND inputIndex < keysPerRound:
            IF Input.GetKeyDown(sequence[inputIndex]):
                inputIndex++
                UpdateQTESuccessEffect(inputIndex)
            ELSE IF Input.anyKeyDown:  // 按键错误
                totalErrors++
                IF totalErrors > maxErrorsAllowed:
                    FinishQTE(false)  // 失败
                    yield break
            
            timeLeft -= Time.deltaTime
            yield return null
        
        // 本轮完成
        currentRound++
        IF timeLeft <= 0:
            totalErrors++
        StartNextRound()
    
    FUNCTION FinishQTE(success: bool):
        qteActive = false
        HideQTEUI()
        ResumeEnemyAI()
        
        endingSystem.qteSuccess = success
        
        IF success:
            Debug.Log("灵魂共鸣仪式完成！")
            PlayQTE_SUCCESS_Effect()
        ELSE:
            Debug.Log("仪式失败...")
            PlayQTE_FAIL_Effect()
```
