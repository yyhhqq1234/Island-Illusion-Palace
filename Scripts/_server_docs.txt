  --- ★ POST /workflows/execute (统一入口) ---
  这是最推荐的方式。自动查找工作流注册表 → 蓝图注册表，智能路由。

  请求:
  {
    "workflow_id": "game-character-design",  // 必填: 工作流或蓝图 ID
    "params": {                                   // 参数, 依工作流而定
      "prompt": "a brave knight",                 // 正向提示词 (大多数需要)
      "negative_prompt": "bad quality, blurry",   // 可选: 负面提示词
      "seed": 42,                                 // 可选: 随机种子
      "steps": 10,                                // 可选: 覆盖默认步数
      "cfg": 7.0,                                 // 可选: CFG scale
      "width": 1024,                              // 可选: 画布宽度
      "height": 1024                              // 可选: 画布高度
    }
  }

  成功响应 (200):
  {
    "prompt_id": "abc123-def456",
    "number": 1,
    "workflow_id": "game-character-design",
    "workflow_name": "Unity 2D 角色精灵",
    "client_id": "random-uuid",
    "node_errors": {}
  }

  蓝图执行成功响应:
  {
    "prompt_id": "xyz789",
    "blueprint_id": "game-item-icon",
    "blueprint_name": "Text to Image (Z-Image-Turbo)",
    "client_id": "random-uuid",
    "node_count": 10
  }

  失败响应 (200, 含 error):
  {"error": "工作流不存在: xxx"}
  {"error": "蓝图不存在: xxx"}
  {"error": "ComfyUI API 调用失败: ..."}

  参数注入规则 (仅注册工作流，服务端自动处理):

  所有工作流统一使用 "prompt" 传参。服务端根据工作流类型自动选择注入方式：

  ┌─ SDXL 工作流 (ID 含 -sdxl) ──────────────────────────────────────┐
  │ prompt:           → CLIPTextEncodeSDXLNPU 的 text_g + text_l       │
  │ negative_prompt:  → 标记"负面"的 CLIPTextEncodeSDXLNPU             │
  │                     (text_g + text_l 同时设置)                       │
  │ ★ 客户端只需传 "prompt"，无需区分 text_g 和 text_l                 │
  └──────────────────────────────────────────────────────────────────┘
  ┌─ SD1.5 工作流 ────────────────────────────────────────────────────┐
  │ prompt:           → CLIPTextEncodeNPU 的 text                      │
  │ negative_prompt:  → 标记"负面"的 CLIPTextEncodeNPU 的 text          │
  └──────────────────────────────────────────────────────────────────┘
  ┌─ 通用参数 (所有工作流) ───────────────────────────────────────────┐
  │ seed:             → KSampler/RandomNoise 的 seed/noise_seed        │
  │ steps:            → KSampler/SamplerCustom 的 steps                │
  │ cfg:              → KSampler 的 cfg                                │
  │ width/height:     → EmptyLatentImage/EmptySD3LatentImage 的对应字段 │
  └──────────────────────────────────────────────────────────────────┘

  3.2 WebSocket API (端口 8189) — 实时通信
  ================================================================

  地址: ws://10.150.164.64:8189

  所有消息格式: {"type": "<类型>", "data": {...}, "timestamp": <Unix秒>}

  === 客户端 → 服务器 ===

  ping               心跳
    发送: {"type":"ping", "data":{}}
    返回: {"type":"pong", "data":{"server_time":1700000000.0}}

  request_status     请求状态 (GPU/队列/建议)
    发送: {"type":"request_status", "data":{}}
    返回: {"type":"server_status", "data":{<同 GET /status>}}

  get_history        获取消息历史
    发送: {"type":"get_history", "data":{"count":50}}
    返回: {"type":"message_history", "data":{"messages":[...]}}

  register           注册客户端元数据
    发送: {"type":"register", "data":{"metadata":{"platform":"unity","version":"2.0"}}}
    返回: {"type":"ack", "data":{"message":"注册成功, ID: client_1_xxx"}}

  chat               广播消息
    发送: {"type":"chat", "data":{"message":"Hello"}}
    广播: {"type":"chat", "data":{"from":"client_1","message":"Hello","time":...}}

  submit_priority    请求优先执行
    发送: {"type":"submit_priority", "data":{"reason":"urgent"}}
    返回: {"type":"ack", "data":{"message":"优先请求已接收"}}

  hot_reload         热更新通知 (内部)
    发送: {"type":"hot_reload", "data":{"message":"...","level":"info"}}
    广播: {"type":"hot_reload", "data":{...}}

  === 工作流管理 (WebSocket) ===

  list_workflows     列出工作流 (可按类别筛选)
    发送: {"type":"list_workflows", "data":{"category":"unity-2d"}}
    返回: {"type":"workflow_list", "data":{<含 total/categories/speed_levels>}}

  get_workflow       获取单个工作流详情 (含 workflow_json)
    发送: {"type":"get_workflow", "data":{"workflow_id":"game-character-design"}}
    返回: {"type":"workflow_detail", "data":{<含 params/workflow_json>}}

  execute_workflow   执行工作流 ★ 统一入口 (自动识别工作流/蓝图)
    发送: {
      "type":"execute_workflow",
      "data":{"workflow_id":"game-character-design", "params":{"prompt":"a knight","seed":42}}
    }
    返回: {"type":"workflow_executed", "data":{"prompt_id":"...", "workflow_name":"...", "number":1}}
    广播: 向其他客户端广播 {"type":"workflow_executed", "data":{...}}

  === 蓝图管理 (WebSocket) ===

  list_blueprints    列出蓝图 (可按类别筛选)

    返回: {"type":"blueprint_list", "data":{"blueprints":{...}, "total":30}}

  welcome            连接成功 (自动)

  === 5.3 WebSocket 方式 (实时监听, 长期运行) ===

  from comms_client import CommsClient
  import time

  client = CommsClient(
      server_url="ws://10.150.164.64:8189",
      auto_reconnect=True,
      ping_interval=30,
      ping_timeout=10
  )

  # —— 注册事件回调 ——

  @client.on("connected")
  def on_connected(data):
      print(f"已连接! client_id: {data['client_id']}")
      client.list_workflows()      # 连接后列出工作流
      client.list_blueprints()     # 列出蓝图

  @client.on("server_status")
  def on_status(data):
      gpu = data["gpu"]
      queue = data["queue"]
      print(f"GPU: {gpu['vram_used']}MB/{gpu['vram_total']}MB | "
            f"队列: {len(queue['pending'])}等待 | "
            f"在线: {data['connected_clients']}")

  @client.on("workflow_list")
  def on_wf_list(data):
      print(f"\n=== 注册工作流 ({data['total']}个) ===")
      for wf_id, wf in data["workflows"].items():
          print(f"  {wf_id}: {wf['name']}")

  @client.on("blueprint_list")
  def on_bp_list(data):
      print(f"\n=== 蓝图 ({data['total']}个) ===")
      cats = {}
      for bp_id, bp in data["blueprints"].items():
          cat = bp.get("category", "other")
          cats.setdefault(cat, []).append(bp['name'])
      for cat, names in cats.items():
          print(f"  [{cat}] {len(names)}个: {names[:3]}...")

  @client.on("blueprint_detail")
  def on_blueprint_detail(data):
      print(f"\n蓝图: {data['name']}")
      print(f"  类别: {data.get('category')}")
      print(f"  参数: {list(data.get('params', {}).keys())}")
      print(f"  输出类型: {data.get('output_type')}")

  @client.on("workflow_executed")
  def on_executed(data):
      wf_name = data.get("workflow_name") or data.get("blueprint_name", "?")
      print(f"任务提交: {wf_name} (prompt_id: {data.get('prompt_id')})")

  @client.on("blueprint_executed")
  def on_bp_executed(data):
      print(f"蓝图提交: {data['blueprint_name']} (nodes: {data['node_count']})")

  @client.on("hot_reload")
  def on_hotreload(data):
      print(f"热更新: [{data['level']}] {data['message']}")

  @client.on("disconnected")
  def on_disconnected(data):
      print(f"断开: {data.get('reason')}, 尝试重连...")

  @client.on("error")
  def on_error(data):
      print(f"服务器错误: {data.get('message')}")

  # —— 连接 ——
  client.connect(blocking=False)
  client.wait_connected(timeout=10)

  # —— 交互 ——
  client.request_status()
  client.list_workflows(category="unity-2d")
  client.list_blueprints(category="text-to-image")
  client.get_workflow("game-character-design")
  client.get_blueprint("image-blur")
  client.execute_workflow("game-concept-art",
      {"prompt": "a dragon flying over mountains, fantasy art"})
  client.execute_blueprint("image-blur",
      {"prompt": "a warrior in battle"})
  client.chat("Pipeline started!")

  # 持续运行
  try:
      while True: time.sleep(1)
  except KeyboardInterrupt:
      client.disconnect()


  === 5.4 获取生成结果 ===

  任务提交后, prompt_id 用于追踪。通过 ComfyUI 原生 API 获取结果。

  import urllib.request, json, time

  def wait_for_result(prompt_id, timeout=300, server="http://10.150.164.64:8189"):
      """轮询等待任务完成，返回输出文件列表"""
      deadline = time.time() + timeout
      while time.time() < deadline:
          # 检查队列 (任务是否还在排队/运行)
          queue = json.loads(urllib.request.urlopen(f"{server}/comfy/queue").read())
          in_queue = False
          for item in queue.get("queue_running", []):
              if item[1] == prompt_id:
                  in_queue = True
                  break
          for item in queue.get("queue_pending", []):
              if item[1] == prompt_id:
                  in_queue = True
                  break

          if not in_queue:
              # 任务已完成 or 不存在 → 查历史
              try:
                  history = json.loads(
                      urllib.request.urlopen(f"{server}/comfy/history/{prompt_id}").read()
                  )
                  if prompt_id in history:
                      outputs = []
                      for node_id, node_out in history[prompt_id].get("outputs", {}).items():
                          for img in node_out.get("images", []):
                              outputs.append({
                                  "filename": img["filename"],
                                  "subfolder": img.get("subfolder", ""),
                                  "type": img.get("type", "output"),
                                  "url": f"{server}/comfy/view?filename={img['filename']}"
                                         f"&subfolder={img.get('subfolder','')}"
                                         f"&type={img.get('type','output')}"
                              })
                          for vid in node_out.get("gifs", []):
                              outputs.append({
                                  "filename": vid["filename"],
                                  "type": "video",
                                  "url": f"{server}/comfy/view?filename={vid['filename']}"
                                         f"&subfolder={vid.get('subfolder','')}"
                                         f"&type=output"
                              })
                      return {"status": "done", "outputs": outputs}
                  else:
                      return {"status": "not_found", "outputs": []}
              except Exception as e:
                  return {"status": "error", "error": str(e), "outputs": []}

          time.sleep(3)

      return {"status": "timeout", "outputs": []}

  # 使用
  result = CommsClient.http_execute_workflow("game-item-icon", {"prompt": "golden sword"})
  prompt_id = result.get("prompt_id")
  if prompt_id:
      outputs = wait_for_result(prompt_id)
      for f in outputs.get("outputs", []):
          print(f"  {f['type']}: {f['filename']} → {f['url']}")

  === 5.6 自动管道 (WebSocket 空闲时提交) ===

  from comms_client import CommsClient
  import time, threading

  client = CommsClient("ws://10.150.164.64:8189")
  tasks = []
  lock = threading.Lock()

  @client.on("server_status")
  def on_status(data):
      gpu = data["gpu"]
      queue = data["queue"]
      # 空闲时提交下一个任务
      if (len(queue["running"]) + len(queue["pending"]) == 0
          and gpu.get("vram_used", 0) < gpu.get("vram_total", 0) * 0.6):
          with lock:
              if tasks:
                  wf_id, params = tasks.pop(0)
                  client.execute_workflow(wf_id, params)
                  print(f"自动提交: {wf_id}")

  @client.on("workflow_executed")
  def on_done(data):
      name = data.get("workflow_name", "")
      print(f"完成: {name}")

  # 添加任务
  tasks.append(("game-concept-art", {"prompt": "a dragon, fantasy art"}))
  tasks.append(("game-environment", {"prompt": "dark castle on cliff, stormy sky"}))
  tasks.append(("game-character-design", {"prompt": "elven archer, green cloak"}))

  client.connect(blocking=False)
  client.wait_connected()

  try:
      while True: time.sleep(1)
  except KeyboardInterrupt:
      client.disconnect()

================================================================================
  9. 错误处理与常见问题
================================================================================

  === 服务端错误 ===

  1. {"error": "工作流不存在: xxx"}
     原因: workflow_id 拼写错误, 或该工作流未在 registry.json 中注册
     解决: GET /workflows 获取正确的 ID 列表

  2. {"error": "蓝图不存在: xxx"}
     原因: blueprint_id 不存在
     解决: GET /blueprints 获取正确 ID, 或 GET /workflows 查全部

  3. {"error": "ComfyUI API 调用失败: ..."}
     原因: ComfyUI 主服务未运行或崩溃
     解决: 等待 monitor_service 自动重启 (最多 180 秒)

  4. {"error": "蓝图展平失败: ..."}
     原因: 蓝图 JSON 格式异常
     解决: 联系管理员检查蓝图文件完整性

  5. HTTP 500 from ComfyUI
     原因: 蓝图需要的模型未安装, 或节点类型不被当前 ComfyUI 识别
     解决: 确认对应模型已安装。部分蓝图依赖特定模型 (如 Flux/Qwen/Z-Image 等),
           这些模型可能需要额外下载

  === 客户端错误 ===

  6. Connection refused / timeout
     原因: 通信服务未启动或网络不通
     解决: curl http://10.150.164.64:8189/health 检查连通性

  7. WebSocket 连接后立即断开
     原因: 网络波动或心跳超时
     解决: SDK 默认开启 auto_reconnect, 会自动重连 (指数退避, 最大30秒)

  8. 任务一直 pending 不执行
     原因: 队列积压
     解决: POST /queue {"delete": ["pending_id"]} 删除等待任务
           或 POST /interrupt 中断当前任务

  === 调试技巧 ===

  检查通信服务:  curl http://10.150.164.64:8189/health
  检查 ComfyUI:   curl http://10.150.164.64:8189/comfy/system_stats
  查看队列:       curl http://10.150.164.64:8189/comfy/queue
  查看历史:       curl http://10.150.164.64:8189/comfy/history?max_items=1
  查看日志:       d:\ComfyUI-P\logs\  (service.log / monitor.log / hotreload.log)

  """comfy_cli.py — 命令行 AI 素材生成工具"""
  from comms_client import CommsClient
  import urllib.request, json, time, sys

  SERVER = "http://10.150.164.64:8189"  # 统一入口

  def list_all():

      print(f"\n=== 注册工作流 ({data['total_workflows']}个) ===")
      for wf_id, wf in data["workflows"]["workflows"].items():
          print(f"  {wf_id:35s} {wf['name']:20s} [{wf['speed']}]")
      print(f"\n=== 蓝图 ({data['total_blueprints']}个) ===")
      cats = {}
      for bp_id, bp in data["blueprints"]["blueprints"].items():
          cats.setdefault(bp.get("category","other"), []).append(bp_id)
      for cat, ids in sorted(cats.items()):
          print(f"  [{cat}] {len(ids)}个: {', '.join(ids[:3])}{'...' if len(ids)>3 else ''}")

  def describe(wf_id):
      # 尝试工作流
      detail = CommsClient.http_get_workflow(wf_id)
      if "error" not in detail:
          print(f"\n工作流: {detail['name']}")
          print(f"  模型: {detail.get('model')} | 画布: {detail.get('canvas')}")
          print(f"  参数: {list(detail.get('params', {}).keys())}")
          return
      # 尝试蓝图
      bp = CommsClient.http_get_blueprint(wf_id)
      if "error" not in bp:
          print(f"\n蓝图: {bp['name']}")
          print(f"  类别: {bp.get('category')} | 输出: {bp.get('output_type')}")
          print(f"  参数: {bp.get('params')}")
          return
      print(f"未找到: {wf_id}")

  def generate(wf_id, *args):
      params = {}
      for a in args:
          if "=" in a:
              k, v = a.split("=", 1)
              params[k] = v
          else:
              params["prompt"] = a  # 默认作为 prompt
      result = CommsClient.http_execute_workflow(wf_id, params)
      if "error" in result:
          print(f"错误: {result['error']}")
          return
      pid = result["prompt_id"]
      name = result.get("workflow_name") or result.get("blueprint_name", wf_id)
      print(f"已提交: {name} (prompt_id: {pid})")

      # 等待完成
      print("等待...", end="", flush=True)
      deadline = time.time() + 300
      while time.time() < deadline:
          q = json.loads(urllib.request.urlopen(f"{SERVER}/comfy/queue").read())
          in_q = False
          for lst in [q.get("queue_running",[]), q.get("queue_pending",[])]:
              for item in lst:
                  if item[1] == pid: in_q = True
          if not in_q:
              print("\n完成!")
              hist = json.loads(urllib.request.urlopen(f"{SERVER}/comfy/history/{pid}").read())
              if pid in hist:
                  for nid, out in hist[pid].get("outputs", {}).items():
                      for img in out.get("images", []):
                          print(f"  文件: {img['filename']}")
                          print(f"  URL:  {SERVER}/comfy/view?filename={img['filename']}&type={img.get('type','output')}")
              return
          print(".", end="", flush=True)
          time.sleep(3)
      print("\n超时!")

      queued = set()
      for lst in [q.get("queue_running",[]), q.get("queue_pending",[])]:
          for item in lst:
              queued.add(item[1])

      for name, pid in list(pending.items()):
          if pid not in queued:
              hist = json.loads(urllib.request.urlopen(
                  f"http://10.150.164.64:8189/comfy/history/{pid}").read())

                          outputs.append(img["filename"])

                  del pending[name]

      if pending:
          print(f"  等待 {len(pending)} 个...")
          time.sleep(5)

  for name, files in results.items():

  client.execute_blueprint(bp_id, params)   # 执行蓝图
